UTILITY FUNCTION: getImageUrl()

FILE: public/js/utils.js
LINES: 92-114

ADDED:
function getImageUrl(path) {
  if (!path || path.trim() === '' || path === 'null' || path === 'undefined') {
    return '';
  }

  if (path.startsWith('http')) {
    return path;
  }

  let finalPath = path;
  
  if (!finalPath.startsWith('/')) {
    finalPath = '/' + finalPath;
  }

  finalPath = finalPath.replace(/\/\//g, '/');

  return window.location.origin + finalPath;
}

BEHAVIOR MATRIX:
  Input: "uploads/events/123/img.jpg"
  Output: "https://localhost:8000/uploads/events/123/img.jpg" ✓
  
  Input: "/uploads/events/123/img.jpg"
  Output: "https://localhost:8000/uploads/events/123/img.jpg" ✓
  
  Input: "https://cdn.example.com/img.jpg"
  Output: "https://cdn.example.com/img.jpg" ✓
  
  Input: null / "" / undefined
  Output: "" ✓
  
  Input: "uploads//events///img.jpg"
  Output: "https://localhost:8000/uploads/events/img.jpg" ✓
